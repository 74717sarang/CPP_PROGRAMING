1.Write a program which accepts a sentence from the user and alters it as follows: Every space is
replaced by *, case of all alphabets is reversed, digits are replaced by?

#include <stdio.h>
#include <ctype.h>
int main() {
    char sentence[1000];
    int i;
    printf("Enter a sentence: ");
    fgets(sentence, sizeof(sentence), stdin);
    for (i = 0; sentence[i] != '\0'; i++) {
        if (sentence[i] == ' ') {
            sentence[i] = '*';
        }
        else if (isalpha(sentence[i])) {
            if (islower(sentence[i])) {
                sentence[i] = toupper(sentence[i]);
            } else {
                sentence[i] = tolower(sentence[i]);
            }
        }
        else if (isdigit(sentence[i])) {
            sentence[i] = '?';
        }
    }
    printf("Altered sentence: %s\n", sentence);
    return 0;
}

2.Write a program that accepts n strings and displays the longest string. Use array of strings.

#include <stdio.h>
#include <string.h>
#define MAX_SIZE 100
int main() {
    char strings[MAX_SIZE][MAX_SIZE];
    int n, maxLength = 0, longestIndex = 0;
    printf("Enter the number of strings: ");
    scanf("%d", &n);
    printf("Enter %d strings:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%s", strings[i]);
    }
    for (int i = 0; i < n; i++) {
        int currentLength = strlen(strings[i]);
        if (currentLength > maxLength) {
            maxLength = currentLength;
            longestIndex = i;
        }
    }
    printf("Longest string: %s\n", strings[longestIndex]);
    return 0;
}

3.Write a menu driven program to perform the following operations on strings using standard library
functions: Length, Copy, Concatenation, Compare, Reverse, Uppercase, Lowercase, Check case.

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#define MAX_LENGTH 100

// Function to display menu
void displayMenu() {
    printf("\nMenu:\n");
    printf("1. Length\n");
    printf("2. Copy\n");
    printf("3. Concatenation\n");
    printf("4. Compare\n");
    printf("5. Reverse\n");
    printf("6. Uppercase\n");
    printf("7. Lowercase\n");
    printf("8. Check Case\n");
    printf("9. Exit\n");
    printf("Enter your choice: ");
}

int main() {
    char str1[MAX_LENGTH], str2[MAX_LENGTH], result[MAX_LENGTH];
    int choice;
    do {
        displayMenu();
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                // Length
                printf("Enter the string: ");
                scanf(" %[^\n]", str1);
                printf("Length of the string: %lu\n", strlen(str1));
                break;
            case 2:
                // Copy
                printf("Enter the source string: ");
                scanf(" %[^\n]", str1);
                strcpy(result, str1);
                printf("Copied string: %s\n", result);
                break;
            case 3:
                // Concatenation
                printf("Enter the first string: ");
                scanf(" %[^\n]", str1);
                printf("Enter the second string: ");
                scanf(" %[^\n]", str2);
                strcat(str1, str2);
                printf("Concatenated string: %s\n", str1);
                break;
            case 4:
                // Compare
                printf("Enter the first string: ");
                scanf(" %[^\n]", str1);
                printf("Enter the second string: ");
                scanf(" %[^\n]", str2);
                if (strcmp(str1, str2) == 0) {
                    printf("Strings are equal.\n");
                } else {
                    printf("Strings are not equal.\n");
                }
                break;
            case 5:
                // Reverse
                printf("Enter the string: ");
                scanf(" %[^\n]", str1);
                strcpy(result, str1);
                strrev(result);
                printf("Reversed string: %s\n", result);
                break;
            case 6:
                // Uppercase
                printf("Enter the string: ");
                scanf(" %[^\n]", str1);
                strcpy(result, str1);
                for (int i = 0; result[i] != '\0'; i++) {
                    result[i] = toupper(result[i]);
                }
                printf("Uppercase string: %s\n", result);
                break;
            case 7:
                // Lowercase
                printf("Enter the string: ");
                scanf(" %[^\n]", str1);
                strcpy(result, str1);
                for (int i = 0; result[i] != '\0'; i++) {
                    result[i] = tolower(result[i]);
                }
                printf("Lowercase string: %s\n", result);
                break;
            case 8:
		//check case
                printf("Enter the character: ");
                scanf(" %c", &str1[0]);
                if (isupper(str1[0])) {
                    printf("Uppercase character.\n");
                } else if (islower(str1[0])) {
                    printf("Lowercase character.\n");
                } else {
                    printf("Not an alphabet character.\n");
                }
                break;
            case 9:
		//exit
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice! Please enter a valid option.\n");
        }
    } while (choice != 9);
    return 0;
}
